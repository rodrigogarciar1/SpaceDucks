from PyQt5 import QtWidgets
from PyQt5.QtWidgets import QApplication, QMainWindow, QLabel, QLineEdit, QVBoxLayout, QWidget
import sys

class MyWindow(QMainWindow):
    def __init__(self):
        super(MyWindow, self).__init__()
        self.setGeometry(200, 200, 600, 600)  # Posición y tamaño de la ventana
        self.setWindowTitle("My App")

        # Inicializar la interfaz
        self.initUI()

    def initUI(self):
        # Crear una etiqueta
        self.label = QLabel("Primer intento", self)
        self.label.move(50, 50)

        # Crear un botón
        self.b1 = QtWidgets.QPushButton("Click", self)
        self.b1.move(50, 100)
        self.b1.clicked.connect(self.click)  # Conectar el clic del botón a la función 'click'

        # Crear un campo de texto (QLineEdit)
        self.input = QLineEdit(self)
        self.input.move(50, 150)
        self.input.textChanged.connect(self.label.setText)  # Conectar el cambio de texto al QLabel

    # Función que se ejecuta al hacer clic en el botón
    def click(self):
        self.label.setText("Has presionado el botón")
        self.update_label()

    # Actualizar el tamaño de la etiqueta
    def update_label(self):
        self.label.adjustSize()

# Inicializar la ventana
def window():
    app = QApplication(sys.argv)
    win = MyWindow()
    win.show()
    sys.exit(app.exec_())

window()
